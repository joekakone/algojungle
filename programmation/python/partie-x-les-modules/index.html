    3. random

    Ce module implémente des générateurs de nombres pseudo-aléatoires pour différentes distributions.

    import random

    # Initialiser le générteur de nombres aléatoires
random.seed(42)

    # Tirer un nombre aléatoire

# Tirer un nombre entre 0 et 1
print(random.random())

# Tirer un nombre entier compris dans un intervalle
print(random.randint(1, 20))

# Tirer un nombre dans une série
print(random.randrange(0, 101, 2))

# Loi normale
print(random.gauss(mu=0, sigma=1))

# Loi uniforme
print(random.uniform(a=2, b=5))

# Loi exponentielle
print(random.expovariate(lambd=1/5))

    0.6394267984578837
1
94
-0.11131586156766246
4.209413642492037
5.645865043992964

    # Tirer un échantillon
colors = ['Green', 'Yellow', 'Black', 'Blue', 'White', 'Pink', 'Red']

# Tirage aléatoire d'un élément
print(random.choice(colors))

# Tirage avec remise
print(random.choices(colors, k=10))

# Tirage sans remise
print(random.sample(colors, k=5))

    White
['Green', 'Black', 'Green', 'Yellow', 'Blue', 'Green', 'Yellow', 'White', 'Blue', 'Yellow']
['White', 'Black', 'Green', 'Yellow', 'Pink']

    # Mélanger les éléments d'une liste
colors = ['Green', 'Yellow', 'Black', 'Blue', 'White', 'Pink', 'Red']

print(colors)

random.shuffle(colors)

print(colors)

    4. datetime

    Le module datetime fournit des classes permettant de manipuler les dates et les heures.

    import datetime

    # La date d'aujourd'hui
print(datetime.date.today())

# Le timestamp actuel
print(datetime.datetime.now())

    2022-10-09
2022-10-09 18:55:11.282400

    # Construire une date
print(datetime.date(year=2022, month=1, day=1))

print(datetime.datetime(year=2022, month=7, day=31, hour=6, minute=45, second=10))

    2022-01-01
2022-07-31 06:45:10

    # Formater une date
date = datetime.date(year=2022, month=12, day=25)

print(date.strftime('%d/%m/%Y'))

print(date.strftime('%A %d %B %Y'))

    25/12/2022
Sunday 25 December 2022

    6. statistics

    Ce module fournit des fonctions pour le calcul de valeurs statistiques sur des données numériques

    import statistics

    # Indicateurs statistiques
values = [1, 2, 2, 4, 5, 6]

# Moyenne
print(statistics.mean(values))

# Médiane
print(statistics.median(values))

# Mode
print(statistics.mode(values))

# Variance
print(statistics.pvariance(values))

# Écart-type
print(statistics.pstdev(values))
